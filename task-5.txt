FONKSIYON YemekKararVer()

    MalzemeEnvanteri = KULLANICIDAN_AL("Evdeki mevcut malzemelerin listesi")
    Bütçe = KULLANICIDAN_AL("Yemek için maksimum bütçe (0 = Evde Pişir)")
    Zaman = KULLANICIDAN_AL("Yemek hazırlığı için maksimum süre (Dakika)")
    RuhHali = KULLANICIDAN_AL("Canınız ne çekiyor? (Hafif, Ağır, Baharatlı, Tatlı, vb.)")

    // *** Koşul: Öncelikli Filtreleme (Bütçe) ***
    
    EĞER Bütçe EŞİT 0 İSE

    DEĞİLSE (Bütçe BÜYÜK 0 İSE)
        EKRANA_YAZDIR("Maksimum bütçe: " + Bütçe + ". Restoran ve tarifler aranıyor.")
        
        // Dışarıdan Yemek Koşulu: Bütçeye uymayan restoranları çıkar
        AdayYemekler = AdayYemekler'den FİYATI Bütçe'den KÜÇÜK Olanları TUT
    SON
    
    // Aday listesinin boş olma ihtimaline karşı kontrol
    EĞER AdayYemekler'in_Uzunluğu EŞİT 0 İSE
        EKRANA_YAZDIR("Üzgünüm, bütçe kısıtlamasına uyan yemek bulunamadı.")
        GERİ_DÖN "Tekrar Deneyin"
    SON
    
    // Sonraki filtrelemeye geç
    Sonuç = FiltrelemeAşaması(AdayYemekler, Kisitlamalar, MalzemeEnvanteri, Zaman, RuhHali)
